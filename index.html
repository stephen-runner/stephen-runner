/* ========================= game.js ========================= */ const BASE_PATH = "/stephen-runner/Assets/"; const ASSET = { stephen: BASE_PATH + "StephenPixel.png", thrower: BASE_PATH + "StoneThrower.png", stone: BASE_PATH + "Stone.png" }; const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); ctx.imageSmoothingEnabled = false; const W = canvas.width; const H = canvas.height; const GROUND_Y = Math.floor(H * 0.78); const player = { x:200, y:GROUND_Y-56, w:56, h:56 }; const images = {}; function load(name, src){ return new Promise(res=>{ const i=new Image(); i.onload=()=>{images[name]=i;res(true)}; i.onerror=()=>res(false); i.src=src; }); } Promise.all([ load("stephen", ASSET.stephen), load("thrower", ASSET.thrower), load("stone", ASSET.stone) ]).then(ok=>{ console.log("Assets loaded:", ok); draw(); }); function draw(){ ctx.clearRect(0,0,W,H); if(images.stephen){ ctx.drawImage(images.stephen, player.x, player.y, player.w, player.h); }
